51static inline bool findRefinement(RefinementComparator& refinementComparator, IThresholdsSubset& thresholdsSubset,
52 const IIndexVector& featureIndices, const IIndexVector& labelIndices,
53 uint32 minCoverage, uint32 numThreads) {
54 bool foundRefinement = false;
55
56 // For each feature, create an object of type `RuleRefinement<RefinementComparator>`...
57 uint32 numFeatures = featureIndices.getNumElements();
58 RuleRefinement<RefinementComparator>* ruleRefinements = new RuleRefinement<RefinementComparator>[numFeatures];
59
60 for (uint32 i = 0; i < numFeatures; i++) {
61 uint32 featureIndex = featureIndices.getIndex(i);
62 RuleRefinement<RefinementComparator>& ruleRefinement = ruleRefinements[i];
63 ruleRefinement.comparatorPtr = std::make_unique<RefinementComparator>(refinementComparator);
64 ruleRefinement.ruleRefinementPtr = labelIndices.createRuleRefinement(thresholdsSubset, featureIndex);
65 }
66
67 // Search for the best condition among all available features to be added to the current rule...
68#pragma omp parallel for firstprivate(numFeatures) firstprivate(ruleRefinements) firstprivate(minCoverage) \
69 schedule(dynamic) num_threads(numThreads)
70 for (int64 i = 0; i < numFeatures; i++) {
71 RuleRefinement<RefinementComparator>& ruleRefinement = ruleRefinements[i];
72 ruleRefinement.ruleRefinementPtr->findRefinement(*ruleRefinement.comparatorPtr, minCoverage);
73 }
74
75 // Pick the best refinement among the refinements that have been found for the different features...
76 for (uint32 i = 0; i < numFeatures; i++) {
77 RuleRefinement<RefinementComparator>& ruleRefinement = ruleRefinements[i];
78 foundRefinement |= refinementComparator.merge(*ruleRefinement.comparatorPtr);
79 }
80
81 delete[] ruleRefinements;
82 return foundRefinement;
83}
