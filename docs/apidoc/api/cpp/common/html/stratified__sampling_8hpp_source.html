19static inline void updateNumExamplesPerLabel(const CContiguousLabelMatrix& labelMatrix, uint32 exampleIndex,
20 uint32* numExamplesPerLabel,
21 std::unordered_map<uint32, uint32>& affectedLabelIndices) {
22 CContiguousLabelMatrix::value_const_iterator labelIterator = labelMatrix.row_values_cbegin(exampleIndex);
23 uint32 numLabels = labelMatrix.getNumCols();
24
25 for (uint32 i = 0; i < numLabels; i++) {
26 if (labelIterator[i]) {
27 uint32 numRemaining = numExamplesPerLabel[i];

48static inline bool tiebreak(uint32 numDesiredSamples, uint32 numDesiredOutOfSamples, RNG& rng) {
49 if (numDesiredSamples > numDesiredOutOfSamples) {
50 return true;
51 } else if (numDesiredSamples < numDesiredOutOfSamples) {
52 return false;
53 } else {
54 return rng.random(0, 2) != 0;
55 }

91 ExampleWiseStratification(const LabelMatrix& labelMatrix, IndexIterator indicesBegin,
92 IndexIterator indicesEnd)
93 : numTotal_(indicesEnd - indicesBegin) {
94 // Create a map that stores the indices of the examples that are associated with each unique label vector...
95 for (uint32 i = 0; i < numTotal_; i++) {
96 uint32 exampleIndex = indicesBegin[i];
97 std::vector<uint32>& exampleIndices = map_[labelMatrix.createView(exampleIndex)];
98 exampleIndices.push_back(exampleIndex);
99 }
100
101 // Sort the label vectors by their frequency...
102 order_.reserve(map_.size());
103
104 for (auto it = map_.begin(); it != map_.end(); it++) {
105 auto& entry = *it;
106 std::vector<uint32>& exampleIndices = entry.second;
107 order_.push_back(exampleIndices);
108 }
109
110 std::sort(order_.begin(), order_.end(), [=](const std::vector<uint32>& a, const std::vector<uint32>& b) {
111 return a.size() < b.size();
112 });
113 }

124 void sampleWeights(BitWeightVector& weightVector, float32 sampleSize, RNG& rng) const {
125 uint32 numTotalSamples = (uint32) std::round(sampleSize * numTotal_);
126 uint32 numTotalOutOfSamples = numTotal_ - numTotalSamples;
127 uint32 numNonZeroWeights = 0;
128 uint32 numZeroWeights = 0;
129
130 for (auto it = order_.begin(); it != order_.end(); it++) {
131 std::vector<uint32>& exampleIndices = *it;
132 std::vector<uint32>::iterator indexIterator = exampleIndices.begin();
133 uint32 numExamples = exampleIndices.size();
134 float32 numSamplesDecimal = sampleSize * numExamples;
135 uint32 numDesiredSamples = numTotalSamples - numNonZeroWeights;
136 uint32 numDesiredOutOfSamples = numTotalOutOfSamples - numZeroWeights;
137 uint32 numSamples = (uint32) (tiebreak(numDesiredSamples, numDesiredOutOfSamples, rng) ?
138 std::ceil(numSamplesDecimal) : std::floor(numSamplesDecimal));
139 numNonZeroWeights += numSamples;
140 numZeroWeights += (numExamples - numSamples);
141
142 // Use the Fisher-Yates shuffle to randomly draw `numSamples` examples and set their weight to 1...
143 uint32 i;
144
145 for (i = 0; i < numSamples; i++) {
146 uint32 randomIndex = rng.random(i, numExamples);
147 uint32 exampleIndex = indexIterator[randomIndex];
148 indexIterator[randomIndex] = indexIterator[i];
149 indexIterator[i] = exampleIndex;
150 weightVector.set(exampleIndex, true);
151 }
152
153 // Set the weights of the remaining examples to 0...
154 for (; i < numExamples; i++) {
155 uint32 exampleIndex = indexIterator[i];
156 weightVector.set(exampleIndex, false);
157 }
158 }
159
160 weightVector.setNumNonZeroWeights(numNonZeroWeights);
161 }

242 LabelWiseStratification(const LabelMatrix& labelMatrix, IndexIterator indicesBegin, IndexIterator indicesEnd)
243 : numRows_(indicesEnd - indicesBegin) {
244 // Convert the given label matrix into the CSC format...
245 const CscLabelMatrix cscLabelMatrix(labelMatrix, indicesBegin, indicesEnd);
246
247 // Create an array that stores for each label the number of examples that are associated with the label, as
248 // well as a sorted map that stores all label indices in increasing order of the number of associated
249 // examples...
250 uint32 numLabels = cscLabelMatrix.getNumCols();
251 uint32* numExamplesPerLabel = new uint32[numLabels];
252 typedef std::set<IndexedValue<uint32>, IndexedValue<uint32>::Compare> SortedSet;
253 SortedSet sortedLabelIndices;
254
255 for (uint32 i = 0; i < numLabels; i++) {
256 uint32 numExamples = cscLabelMatrix.column_indices_cend(i) - cscLabelMatrix.column_indices_cbegin(i);
257 numExamplesPerLabel[i] = numExamples;
258
259 if (numExamples > 0) {
260 sortedLabelIndices.emplace(i, numExamples);
261 }
262 }
263
264 // Allocate arrays for storing the row and column indices of the labels to be processed by the sampling
265 // method in the CSC format...
266 rowIndices_ = (uint32*) malloc(cscLabelMatrix.getNumNonZeroElements() * sizeof(uint32));
267 colIndices_ = (uint32*) malloc((sortedLabelIndices.size() + 1) * sizeof(uint32));
268 uint32 numNonZeroElements = 0;
269 uint32 numCols = 0;
270
271 // Create a boolean array that stores whether individual examples remain to be processed (1) or not (0)...
272 uint32 numTotalExamples = labelMatrix.getNumRows();
273 uint8* mask = new uint8[numTotalExamples] {};
274
275 for (uint32 i = 0; i < numRows_; i++) {
276 uint32 exampleIndex = indicesBegin[i];
277 mask[exampleIndex] = 1;
278 }
279
280 // As long as there are labels that have not been processed yet, proceed with the label that has the
281 // smallest number of associated examples...
282 std::unordered_map<uint32, uint32> affectedLabelIndices;
283 SortedSet::iterator firstEntry;
284
285 while ((firstEntry = sortedLabelIndices.begin()) != sortedLabelIndices.end()) {
286 const IndexedValue<uint32>& entry = *firstEntry;
287 uint32 labelIndex = entry.index;
288
289 // Remove the label from the sorted map...
290 sortedLabelIndices.erase(firstEntry);
291
292 // Add the number of non-zero labels that have been processed so far to the array of column indices...
293 colIndices_[numCols] = numNonZeroElements;
294 numCols++;
295
296 // Iterate the examples that are associated with the current label, if no weight has been set yet...
297 CscLabelMatrix::index_const_iterator indexIterator = cscLabelMatrix.column_indices_cbegin(labelIndex);
298 uint32 numExamples = cscLabelMatrix.column_indices_cend(labelIndex) - indexIterator;
299
300 for (uint32 i = 0; i < numExamples; i++) {
301 uint32 exampleIndex = indexIterator[i];
302
303 // If the example has not been processed yet...
304 if (mask[exampleIndex]) {
305 mask[exampleIndex] = 0;
306
307 // Add the example's index to the array of row indices...
308 rowIndices_[numNonZeroElements] = exampleIndex;
309 numNonZeroElements++;
310
311 // For each label that is associated with the example, decrement the number of associated
312 // examples by one...
313 updateNumExamplesPerLabel(labelMatrix, exampleIndex, &numExamplesPerLabel[0],
314 affectedLabelIndices);
315 }
316 }
317
318 // Remove each label, for which the number of associated examples have been changed previously, from the
319 // sorted map and add it again to update the order...
320 for (auto it = affectedLabelIndices.cbegin(); it != affectedLabelIndices.cend(); it++) {
321 uint32 key = it->first;
322
323 if (key != labelIndex) {
324 uint32 value = it->second;
325 SortedSet::iterator it2 = sortedLabelIndices.find(IndexedValue<uint32>(key, value));
326 uint32 numRemaining = numExamplesPerLabel[key];
327
328 if (numRemaining > 0) {
329 sortedLabelIndices.emplace_hint(it2, key, numRemaining);
330 }
331
332 sortedLabelIndices.erase(it2);
333 }
334 }
335
336 affectedLabelIndices.clear();
337 }
338
339 // If there are examples that are not associated with any labels, we handle them separately..
340 uint32 numRemaining = numRows_ - numNonZeroElements;
341
342 if (numRemaining > 0) {
343 // Adjust the size of the arrays that are used to store row and column indices...
344 rowIndices_ = (uint32*) realloc(rowIndices_, (numNonZeroElements + numRemaining) * sizeof(uint32));
345 colIndices_ = (uint32*) realloc(colIndices_, (numCols + 1) * sizeof(uint32));
346
347 // Add the number of non-zero labels that have been processed so far to the array of column indices...
348 colIndices_[numCols] = numNonZeroElements;
349 numCols++;
350
351 // Iterate the weights of all examples to find those whose weight has not been set yet...
352 for (uint32 i = 0; i < numTotalExamples; i++) {
353 if (mask[i]) {
354 // Add the example's index to the array of row indices...
355 rowIndices_[numNonZeroElements] = i;
356 numNonZeroElements++;
357 }
358 }
359 } else {
360 // Adjust the size of the arrays that are used to store row and column indices...
361 rowIndices_ = (uint32*) realloc(rowIndices_, numNonZeroElements * sizeof(uint32));
362 colIndices_ = (uint32*) realloc(colIndices_, numCols * sizeof(uint32));
363 }
364
365 colIndices_[numCols - 1] = numNonZeroElements;
366 numCols_ = numCols - 1;
367
368 delete[] numExamplesPerLabel;
369 delete[] mask;
370 }
