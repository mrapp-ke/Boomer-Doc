59 void induceDefaultRule(IStatistics& statistics, IModelBuilder& modelBuilder) const override final {
60 uint32 numStatistics = statistics.getNumStatistics();
61 uint32 numLabels = statistics.getNumLabels();
62 CompleteIndexVector labelIndices(numLabels);
63 EqualWeightVector weights(numStatistics);
64 std::unique_ptr<IStatisticsSubset> statisticsSubsetPtr = statistics.createSubset(labelIndices, weights);
65
66 for (uint32 i = 0; i < numStatistics; i++) {
67 statisticsSubsetPtr->addToSubset(i);
68 }
69
70 const IScoreVector& scoreVector = statisticsSubsetPtr->calculateScores();
71 std::unique_ptr<AbstractEvaluatedPrediction> defaultPredictionPtr;
72 ScoreProcessor scoreProcessor(defaultPredictionPtr);
73 scoreProcessor.processScores(scoreVector);
74
75 for (uint32 i = 0; i < numStatistics; i++) {
76 defaultPredictionPtr->apply(statistics, i);
77 }
78
79 modelBuilder.setDefaultRule(defaultPredictionPtr);

82 bool induceRule(IThresholds& thresholds, const IIndexVector& labelIndices, const IWeightVector& weights,
83 IPartition& partition, IFeatureSampling& featureSampling, const IRulePruning& rulePruning,
84 const IPostProcessor& postProcessor, RNG& rng,
85 IModelBuilder& modelBuilder) const override final {
86 std::unique_ptr<ConditionList> conditionListPtr;
87 std::unique_ptr<AbstractEvaluatedPrediction> headPtr;
88 std::unique_ptr<IThresholdsSubset> thresholdsSubsetPtr = this->growRule(
89 thresholds, labelIndices, weights, partition, featureSampling, rng, conditionListPtr, headPtr);
90
91 if (headPtr) {
92 if (weights.hasZeroWeights()) {
93 // Prune rule...
94 IStatisticsProvider& statisticsProvider = thresholds.getStatisticsProvider();
95 statisticsProvider.switchToPruningRuleEvaluation();
96 std::unique_ptr<ICoverageState> coverageStatePtr =
97 rulePruning.prune(*thresholdsSubsetPtr, partition, *conditionListPtr, *headPtr);
98 statisticsProvider.switchToRegularRuleEvaluation();
99
100 // Re-calculate the scores in the head based on the entire training data...
101 if (recalculatePredictions_) {
102 const ICoverageState& coverageState =
103 coverageStatePtr ? *coverageStatePtr : thresholdsSubsetPtr->getCoverageState();
104 partition.recalculatePrediction(*thresholdsSubsetPtr, coverageState, *headPtr);
105 }
106 }
107
108 // Apply post-processor...
109 postProcessor.postProcess(*headPtr);
110
111 // Update the statistics by applying the predictions of the new rule...
112 thresholdsSubsetPtr->applyPrediction(*headPtr);
113
114 // Add the induced rule to the model...
115 modelBuilder.addRule(conditionListPtr, headPtr);
116 return true;
117 } else {
118 // No rule could be induced, because no useful condition could be found. This might be the case, if all
119 // examples have the same values for the considered features.
120 return false;
121 }
122 }
