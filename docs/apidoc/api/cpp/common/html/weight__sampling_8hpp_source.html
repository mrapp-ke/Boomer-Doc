23template<typename Iterator>
24static inline void sampleWeightsWithoutReplacementViaTrackingSelection(BitWeightVector& weightVector, Iterator iterator,
25 uint32 numTotal, uint32 numSamples, RNG& rng) {
26 weightVector.clear();
27 std::unordered_set<uint32> selectedIndices;
28
29 for (uint32 i = 0; i < numSamples; i++) {
30 bool shouldContinue = true;
31 uint32 sampledIndex;
32
33 while (shouldContinue) {
34 uint32 randomIndex = rng.random(0, numTotal);
35 sampledIndex = iterator[randomIndex];
36 shouldContinue = !selectedIndices.insert(sampledIndex).second;
37 }
38
39 weightVector.set(sampledIndex, true);
40 }
41
42 weightVector.setNumNonZeroWeights(numSamples);
43}

57template<typename Iterator>
58static inline void sampleWeightsWithoutReplacementViaPool(BitWeightVector& weightVector, Iterator iterator,
59 uint32 numTotal, uint32 numSamples, RNG& rng) {
60 weightVector.clear();
61 uint32* pool = new uint32[numTotal];
62
63 // Initialize pool...
64 for (uint32 i = 0; i < numTotal; i++) {
65 pool[i] = iterator[i];
66 }
67
68 for (uint32 i = 0; i < numSamples; i++) {
69 // Randomly select an index that has not been drawn yet...
70 uint32 randomIndex = rng.random(0, numTotal - i);
71 uint32 sampledIndex = pool[randomIndex];
72
73 // Set weight at the selected index to 1...
74 weightVector.set(sampledIndex, true);
75
76 // Move the index at the border to the position of the recently drawn index...
77 pool[randomIndex] = pool[numTotal - i - 1];
78 }
79
80 delete[] pool;
81 weightVector.setNumNonZeroWeights(numSamples);
82}
