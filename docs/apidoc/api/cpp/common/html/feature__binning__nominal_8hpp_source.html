38 std::unordered_map<float32, uint32> mapping;

43 thresholdIterator[0] = 0;
44 nextBinIndex++;
45 }
46

48 float32 currentValue = featureIterator[i].value;

51 uint32 index = featureIterator[i].index;
52 auto mapIterator = mapping.emplace(currentValue, nextBinIndex);
53
54 if (mapIterator.second) {
