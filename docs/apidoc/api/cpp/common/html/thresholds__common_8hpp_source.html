54static inline Quality evaluateOutOfSampleInternally(const WeightVector& weights, const CoverageSet& coverageSet,
55 BiPartition& partition, const IStatistics& statistics,
56 const AbstractPrediction& prediction) {
57 OutOfSampleWeightVector<WeightVector> outOfSampleWeights(weights);
58 std::unique_ptr<IStatisticsSubset> statisticsSubsetPtr =
59 prediction.createStatisticsSubset(statistics, outOfSampleWeights);
60 uint32 numCovered = coverageSet.getNumCovered();
61 CoverageSet::const_iterator coverageSetIterator = coverageSet.cbegin();
62 partition.sortSecond();
63 auto holdoutSetIterator = make_binary_forward_iterator(partition.second_cbegin(), partition.second_cend());
64 uint32 previousExampleIndex = 0;
65
66 for (uint32 i = 0; i < numCovered; i++) {
67 uint32 exampleIndex = coverageSetIterator[i];
68 std::advance(holdoutSetIterator, exampleIndex - previousExampleIndex);
69
70 if (*holdoutSetIterator && statisticsSubsetPtr->hasNonZeroWeight(exampleIndex)) {
71 statisticsSubsetPtr->addToSubset(exampleIndex);
72 }
73
74 previousExampleIndex = exampleIndex;
75 }
76
77 return statisticsSubsetPtr->calculateScores();
78}
