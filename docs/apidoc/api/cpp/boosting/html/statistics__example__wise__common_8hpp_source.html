13 void applyExampleWisePredictionInternally(uint32 statisticIndex, const Prediction& prediction,
14 const LabelMatrix& labelMatrix, StatisticView& statisticView,
15 ScoreMatrix& scoreMatrix, const LossFunction& lossFunction) {
16 // Update the scores that are currently predicted for the example at the given index...
17 scoreMatrix.addToRowFromSubset(statisticIndex, prediction.scores_cbegin(), prediction.scores_cend(),
18 prediction.indices_cbegin(), prediction.indices_cend());
19
20 // Update the gradients and Hessians for the example at the given index...
21 lossFunction.updateExampleWiseStatistics(statisticIndex, labelMatrix, scoreMatrix, statisticView);
22 }

124 void resetSubset() override final {
125 if (!accumulatedSumVector_) {
126 // Create a vector for storing the accumulated sums of gradients and Hessians, if
127 // necessary...
128 accumulatedSumVector_ = new StatisticVector(sumVector_);
129 } else {
130 // Add the sum of gradients and Hessians to the accumulated sums of gradients and
131 // Hessians...
132 accumulatedSumVector_->add(sumVector_.gradients_cbegin(), sumVector_.gradients_cend(),
133 sumVector_.hessians_cbegin(), sumVector_.hessians_cend());
134 }

136 // Reset the sum of gradients and Hessians to zero...
137 sumVector_.clear();

143 const IScoreVector& calculatePrediction(bool uncovered, bool accumulated) override final {
144 StatisticVector& sumsOfStatistics = accumulated ? *accumulatedSumVector_ : sumVector_;
145
146 if (uncovered) {
147 tmpVector_.difference(totalSumVector_->gradients_cbegin(),
148 totalSumVector_->gradients_cend(), totalSumVector_->hessians_cbegin(),
149 totalSumVector_->hessians_cend(), labelIndices_,
150 sumsOfStatistics.gradients_cbegin(),
151 sumsOfStatistics.gradients_cend(), sumsOfStatistics.hessians_cbegin(),
152 sumsOfStatistics.hessians_cend());
153 return ruleEvaluationPtr_->calculatePrediction(tmpVector_);
154 }

156 return ruleEvaluationPtr_->calculatePrediction(sumsOfStatistics);
157 }
