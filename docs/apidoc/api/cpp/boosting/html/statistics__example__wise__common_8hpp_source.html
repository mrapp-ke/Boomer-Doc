13 void applyExampleWisePredictionInternally(uint32 statisticIndex, const Prediction& prediction,
14 const LabelMatrix& labelMatrix, StatisticView& statisticView,
15 ScoreMatrix& scoreMatrix, const LossFunction& lossFunction) {
16 // Update the scores that are currently predicted for the example at the given index...
17 scoreMatrix.addToRowFromSubset(statisticIndex, prediction.scores_cbegin(), prediction.scores_cend(),
18 prediction.indices_cbegin(), prediction.indices_cend());
19
20 // Update the gradients and Hessians for the example at the given index...
21 lossFunction.updateExampleWiseStatistics(statisticIndex, labelMatrix, scoreMatrix, statisticView);
22 }

103 void addToMissing(uint32 statisticIndex, float64 weight) override {
104 // Create a vector for storing the totals sums of gradients and Hessians, if necessary...
105 if (totalCoverableSumVector_ == nullptr) {
106 totalCoverableSumVector_ = new StatisticVector(*totalSumVector_);
107 totalSumVector_ = totalCoverableSumVector_;
108 }
109
110 // Subtract the gradients and Hessians of the example at the given index (weighted by the given
111 // weight) from the total sums of gradients and Hessians...
112 totalCoverableSumVector_->add(
113 statistics_.statisticViewPtr_->gradients_row_cbegin(statisticIndex),
114 statistics_.statisticViewPtr_->gradients_row_cend(statisticIndex),
115 statistics_.statisticViewPtr_->hessians_row_cbegin(statisticIndex),
116 statistics_.statisticViewPtr_->hessians_row_cend(statisticIndex), -weight);
117 }

122 void addToSubset(uint32 statisticIndex, float64 weight) override {
123 sumVector_.addToSubset(statistics_.statisticViewPtr_->gradients_row_cbegin(statisticIndex),
124 statistics_.statisticViewPtr_->gradients_row_cend(statisticIndex),
125 statistics_.statisticViewPtr_->hessians_row_cbegin(statisticIndex),
126 statistics_.statisticViewPtr_->hessians_row_cend(statisticIndex),
127 labelIndices_, weight);
128 }
129
133 void resetSubset() override {
134 // Create a vector for storing the accumulated sums of gradients and Hessians, if necessary...
135 if (accumulatedSumVector_ == nullptr) {
136 uint32 numPredictions = labelIndices_.getNumElements();
137 accumulatedSumVector_ = new StatisticVector(numPredictions, true);
138 }
139
140 // Reset the sum of gradients and Hessians to zero and add it to the accumulated sums of
141 // gradients and Hessians...
142 accumulatedSumVector_->add(sumVector_.gradients_cbegin(), sumVector_.gradients_cend(),
143 sumVector_.hessians_cbegin(), sumVector_.hessians_cend());
144 sumVector_.clear();
145 }
146
150 const IScoreVector& calculatePrediction(bool uncovered, bool accumulated) override {
151 StatisticVector& sumsOfStatistics = accumulated ? *accumulatedSumVector_ : sumVector_;
152
153 if (uncovered) {
154 tmpVector_.difference(totalSumVector_->gradients_cbegin(),
155 totalSumVector_->gradients_cend(), totalSumVector_->hessians_cbegin(),
156 totalSumVector_->hessians_cend(), labelIndices_,
157 sumsOfStatistics.gradients_cbegin(),
158 sumsOfStatistics.gradients_cend(), sumsOfStatistics.hessians_cbegin(),
159 sumsOfStatistics.hessians_cend());
160 return ruleEvaluationPtr_->calculatePrediction(tmpVector_);
161 }
162
163 return ruleEvaluationPtr_->calculatePrediction(sumsOfStatistics);
164 }
