184 const IndexVector& labelIndices)
185 : ExampleWiseStatisticsSubset<StatisticVector, StatisticView, RuleEvaluationFactory,

195 void resetSubset() override final {
196 if (!accumulatedSumVectorPtr_) {
197 // Create a vector for storing the accumulated sums of gradients and Hessians, if
198 // necessary...
199 accumulatedSumVectorPtr_ = std::make_unique<StatisticVector>(this->sumVector_);
200 } else {
201 // Add the sums of gradients and Hessians to the accumulated sums of gradients and
202 // Hessians...
203 accumulatedSumVectorPtr_->add(
204 this->sumVector_.gradients_cbegin(), this->sumVector_.gradients_cend(),
205 this->sumVector_.hessians_cbegin(), this->sumVector_.hessians_cend());
206 }
207
208 // Reset the sums of gradients and Hessians to zero...
209 this->sumVector_.clear();
210 }
211
215 const IScoreVector& calculateScoresAccumulated() override final {
216 return this->ruleEvaluationPtr_->calculateScores(*accumulatedSumVectorPtr_);
217 }
218
222 const IScoreVector& calculateScoresUncovered() override final {
223 tmpVector_.difference(totalSumVector_->gradients_cbegin(), totalSumVector_->gradients_cend(),
224 totalSumVector_->hessians_cbegin(), totalSumVector_->hessians_cend(),
225 this->labelIndices_, this->sumVector_.gradients_cbegin(),
226 this->sumVector_.gradients_cend(), this->sumVector_.hessians_cbegin(),
227 this->sumVector_.hessians_cend());
228 return this->ruleEvaluationPtr_->calculateScores(tmpVector_);
229 }
230
234 const IScoreVector& calculateScoresUncoveredAccumulated() override final {
235 tmpVector_.difference(
236 totalSumVector_->gradients_cbegin(), totalSumVector_->gradients_cend(),
237 totalSumVector_->hessians_cbegin(), totalSumVector_->hessians_cend(), this->labelIndices_,
238 accumulatedSumVectorPtr_->gradients_cbegin(), accumulatedSumVectorPtr_->gradients_cend(),
239 accumulatedSumVectorPtr_->hessians_cbegin(), accumulatedSumVectorPtr_->hessians_cend());
240 return this->ruleEvaluationPtr_->calculateScores(tmpVector_);
241 }
