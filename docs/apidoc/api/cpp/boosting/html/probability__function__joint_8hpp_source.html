66 virtual std::unique_ptr<DenseVector<float64>> transformScoresIntoJointProbabilities(
67 const LabelVectorSet& labelVectorSet, VectorConstView<float64>::const_iterator scoresBegin,
68 VectorConstView<float64>::const_iterator scoresEnd) const {
69 uint32 numLabelVectors = labelVectorSet.getNumLabelVectors();
70 std::unique_ptr<DenseVector<float64>> jointProbabilityVectorPtr =
71 std::make_unique<DenseVector<float64>>(numLabelVectors);
72 DenseVector<float64>::iterator jointProbabilityIterator = jointProbabilityVectorPtr->begin();
73 float64 sumOfJointProbabilities = 0;
74
75 // Calculate joint probabilities...
76 LabelVectorSet::const_iterator labelVectorIterator = labelVectorSet.cbegin();
77
78 for (uint32 i = 0; i < numLabelVectors; i++) {
79 const LabelVector& labelVector = *labelVectorIterator[i];
80 float64 jointProbability =
81 this->transformScoresIntoJointProbability(i, labelVector, scoresBegin, scoresEnd);
82 sumOfJointProbabilities += jointProbability;
83 jointProbabilityIterator[i] = jointProbability;
84 }
85
86 // Normalize joint probabilities...
87 for (uint32 i = 0; i < numLabelVectors; i++) {
88 float64 jointProbability = jointProbabilityIterator[i];
89 jointProbabilityIterator[i] = divideOrZero(jointProbability, sumOfJointProbabilities);
90 }
91
92 return jointProbabilityVectorPtr;
93 }
