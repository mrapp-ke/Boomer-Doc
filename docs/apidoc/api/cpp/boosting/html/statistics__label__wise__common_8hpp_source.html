191 void resetSubset() override final {
192 if (!accumulatedSumVectorPtr_) {
193 // Create a vector for storing the accumulated sums of gradients and Hessians, if
194 // necessary...
195 accumulatedSumVectorPtr_ = std::make_unique<StatisticVector>(this->sumVector_);
196 } else {
197 // Add the sums of gradients and Hessians to the accumulated sums of gradients and

199 accumulatedSumVectorPtr_->add(this->sumVector_);

202 // Reset the sums of gradients and Hessians to zero...
203 this->sumVector_.clear();
204 }
