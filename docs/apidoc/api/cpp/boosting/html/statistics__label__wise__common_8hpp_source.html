11 template<typename Prediction, typename LabelMatrix, typename StatisticView, typename ScoreMatrix,
12 typename LossFunction>
13 static inline void applyLabelWisePredictionInternally(uint32 statisticIndex, const Prediction& prediction,
14 const LabelMatrix& labelMatrix, StatisticView& statisticView,
15 ScoreMatrix& scoreMatrix, const LossFunction& lossFunction) {
16 // Update the scores that are currently predicted for the example at the given index...
17 scoreMatrix.addToRowFromSubset(statisticIndex, prediction.scores_cbegin(), prediction.scores_cend(),
18 prediction.indices_cbegin(), prediction.indices_cend());
19
20 // Update the gradients and Hessians of the example at the given index...
21 lossFunction.updateLabelWiseStatistics(statisticIndex, labelMatrix, scoreMatrix, prediction.indices_cbegin(),
22 prediction.indices_cend(), statisticView);
23 }

103 void addToMissing(uint32 statisticIndex, float64 weight) override {
104 // Create a vector for storing the totals sums of gradients and Hessians, if necessary...
105 if (!totalCoverableSumVector_) {
106 totalCoverableSumVector_ = new StatisticVector(*totalSumVector_);
107 totalSumVector_ = totalCoverableSumVector_;
108 }
109
110 // Subtract the gradients and Hessians of the example at the given index (weighted by the given
111 // weight) from the total sums of gradients and Hessians...
112 totalCoverableSumVector_->add(statistics_.statisticViewPtr_->row_cbegin(statisticIndex),
113 statistics_.statisticViewPtr_->row_cend(statisticIndex), -weight);
114 }

119 void addToSubset(uint32 statisticIndex, float64 weight) override {
120 sumVector_.addToSubset(statistics_.statisticViewPtr_->row_cbegin(statisticIndex),
121 statistics_.statisticViewPtr_->row_cend(statisticIndex), labelIndices_,
122 weight);
123 }
124
128 void resetSubset() override {
129 // Create a vector for storing the accumulated sums of gradients and Hessians, if necessary...
130 if (!accumulatedSumVector_) {
131 uint32 numPredictions = labelIndices_.getNumElements();
132 accumulatedSumVector_ = new StatisticVector(numPredictions, true);
133 }
134
135 // Reset the sums of gradients and Hessians to zero and add it to the accumulated sums of
136 // gradients and Hessians...
137 accumulatedSumVector_->add(sumVector_.cbegin(), sumVector_.cend());
138 sumVector_.clear();
139 }

144 const IScoreVector& calculatePrediction(bool uncovered, bool accumulated) override {
145 StatisticVector& sumsOfStatistics = accumulated ? *accumulatedSumVector_ : sumVector_;
146
147 if (uncovered) {
148 tmpVector_.difference(totalSumVector_->cbegin(), totalSumVector_->cend(), labelIndices_,
149 sumsOfStatistics.cbegin(), sumsOfStatistics.cend());
150 return ruleEvaluationPtr_->calculatePrediction(tmpVector_);
151 }
152
153 return ruleEvaluationPtr_->calculatePrediction(sumsOfStatistics);
154 }
