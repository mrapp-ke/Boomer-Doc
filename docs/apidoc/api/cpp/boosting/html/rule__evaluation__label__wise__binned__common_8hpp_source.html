26 static inline float64 calculateBinnedScores(DenseLabelWiseStatisticVector::const_iterator statisticIterator,
27 ScoreIterator scoreIterator, const uint32* weights, uint32 numElements,
28 float64 l1RegularizationWeight, float64 l2RegularizationWeight) {
29 float64 quality = 0;
30
31 for (uint32 i = 0; i < numElements; i++) {
32 uint32 weight = weights[i];
33 const Tuple<float64>& tuple = statisticIterator[i];
34 float64 predictedScore = calculateLabelWiseScore(tuple.first, tuple.second, weight * l1RegularizationWeight,
35 weight * l2RegularizationWeight);
36 scoreIterator[i] = predictedScore;
37 quality += calculateLabelWiseQuality(predictedScore, tuple.first, tuple.second,
38 weight * l1RegularizationWeight, weight * l2RegularizationWeight);
39 }
40
41 return quality;
42 }

127 const IScoreVector& calculateScores(StatisticVector& statisticVector) override final {
128 // Calculate label-wise criteria...
129 uint32 numCriteria =
130 this->calculateLabelWiseCriteria(statisticVector, criteria_, scoreVector_.getNumElements(),
131 l1RegularizationWeight_, l2RegularizationWeight_);
132
133 // Obtain information about the bins to be used...
134 LabelInfo labelInfo = binningPtr_->getLabelInfo(criteria_, numCriteria);
135 uint32 numBins = labelInfo.numPositiveBins + labelInfo.numNegativeBins;
136 scoreVector_.setNumBins(numBins, false);
137
138 // Reset arrays to zero...
139 DenseLabelWiseStatisticVector::iterator aggregatedStatisticIterator =
140 aggregatedStatisticVector_.begin();
141 setArrayToZeros(aggregatedStatisticIterator, numBins);
142 setArrayToZeros(numElementsPerBin_, numBins);
143
144 // Apply binning method in order to aggregate the gradients and Hessians that belong to the same bins...
145 typename StatisticVector::const_iterator statisticIterator = statisticVector.cbegin();
146 typename DenseBinnedScoreVector<IndexVector>::index_binned_iterator binIndexIterator =
147 scoreVector_.indices_binned_begin();
148 auto callback = [=](uint32 binIndex, uint32 labelIndex) {
149 aggregatedStatisticIterator[binIndex] += statisticIterator[labelIndex];
150 numElementsPerBin_[binIndex] += 1;
151 binIndexIterator[labelIndex] = binIndex;
152 };
153 auto zeroCallback = [=](uint32 labelIndex) {
154 binIndexIterator[labelIndex] = maxBins_;
155 };
156 binningPtr_->createBins(labelInfo, criteria_, numCriteria, callback, zeroCallback);
157
158 // Compute predictions, as well as their overall quality...
159 typename DenseBinnedScoreVector<IndexVector>::score_binned_iterator scoreIterator =
160 scoreVector_.scores_binned_begin();
161 scoreVector_.quality =
162 calculateBinnedScores(aggregatedStatisticIterator, scoreIterator, numElementsPerBin_, numBins,
163 l1RegularizationWeight_, l2RegularizationWeight_);
164 return scoreVector_;
165 }
