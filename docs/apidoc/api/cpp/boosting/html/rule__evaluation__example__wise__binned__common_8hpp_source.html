23 static inline uint32 removeEmptyBins(uint32* numElementsPerBin, uint32* binIndices, uint32 numBins) {
24 uint32 n = 0;
25
26 for (uint32 i = 0; i < numBins; i++) {
27 binIndices[i] = n;
28 uint32 numElements = numElementsPerBin[i];
29
30 if (numElements > 0) {
31 numElementsPerBin[n] = numElements;
32 n++;
33 }
34 }
35

57 template<typename BinIndexIterator>
58 static inline void aggregateGradientsAndHessians(
59 DenseExampleWiseStatisticVector::gradient_const_iterator gradientIterator,
60 DenseExampleWiseStatisticVector::hessian_const_iterator hessianIterator, uint32 numElements,
61 BinIndexIterator binIndexIterator, const uint32* binIndices, float64* gradients, float64* hessians,
62 uint32 maxBins) {
63 for (uint32 i = 0; i < numElements; i++) {
64 uint32 originalBinIndex = binIndexIterator[i];
65
66 if (originalBinIndex != maxBins) {
67 uint32 binIndex = binIndices[originalBinIndex];
68 binIndexIterator[i] = binIndex;
69
70 // Add the gradient that corresponds to the `i`-th element of the original gradient vector to the
71 // corresponding element of the aggregated gradient vector...
72 gradients[binIndex] += gradientIterator[i];
73
74 // Add the Hessian that corresponds to the `i`-th element on the diagonal of the original Hessian matrix
75 // to the corresponding element of the aggregated Hessian matrix...
76 hessians[triangularNumber(binIndex + 1) - 1] += hessianIterator[triangularNumber(i + 1) - 1];
77 }
78 }
79
80 for (uint32 i = 1; i < numElements; i++) {
81 uint32 binIndex = binIndexIterator[i];
82
83 if (binIndex != maxBins) {
84 for (uint32 j = 0; j < i; j++) {
85 uint32 binIndex2 = binIndexIterator[j];
86
87 // Add the hessian at the `i`-th row and `j`-th column of the original Hessian matrix to the
88 // corresponding element of the aggregated Hessian matrix, if the labels at indices `i` and `j` do
89 // not belong to the same bin...
90 if (binIndex2 != maxBins && binIndex != binIndex2) {
91 uint32 r, c;
92
93 if (binIndex < binIndex2) {
94 r = binIndex;
95 c = binIndex2;
96 } else {
97 r = binIndex2;
98 c = binIndex;
99 }
100
101 hessians[triangularNumber(c) + r] += hessianIterator[triangularNumber(i) + j];
102 }
103 }
104 }
105 }
106 }

118 static inline void addL1RegularizationWeight(float64* ordinates, uint32 n, const uint32* weights,
119 float64 l1RegularizationWeight) {
120 for (uint32 i = 0; i < n; i++) {
121 uint32 weight = weights[i];
122 float64 gradient = ordinates[i];
123 ordinates[i] += (weight * getL1RegularizationWeight(gradient, l1RegularizationWeight));
124 }

137 static inline void addL2RegularizationWeight(float64* coefficients, uint32 numPredictions, const uint32* weights,
138 float64 l2RegularizationWeight) {
139 for (uint32 i = 0; i < numPredictions; i++) {
140 uint32 weight = weights[i];
141 coefficients[(i * numPredictions) + i] += (weight * l2RegularizationWeight);
142 }

156 template<typename ScoreIterator>
157 static inline float64 calculateRegularizationTerm(ScoreIterator scores, const uint32* numElementsPerBin,
158 uint32 numBins, float64 l1RegularizationWeight,
159 float64 l2RegularizationWeight) {
160 float64 regularizationTerm;
161
162 if (l1RegularizationWeight > 0) {
163 regularizationTerm = l1RegularizationWeight * l1Norm(scores, numElementsPerBin, numBins);
164 } else {
165 regularizationTerm = 0;
166 }
167
168 if (l2RegularizationWeight > 0) {
169 regularizationTerm += 0.5 * l2RegularizationWeight * l2NormPow(scores, numElementsPerBin, numBins);
170 }
171
172 return regularizationTerm;
173 }

278 const IScoreVector& calculateScores(DenseExampleWiseStatisticVector& statisticVector) override final {
279 // Calculate label-wise criteria...
280 uint32 numCriteria =
281 this->calculateLabelWiseCriteria(statisticVector, criteria_, scoreVector_.getNumElements(),
282 l1RegularizationWeight_, l2RegularizationWeight_);
283
284 // Obtain information about the bins to be used...
285 LabelInfo labelInfo = binningPtr_->getLabelInfo(criteria_, numCriteria);
286 uint32 numBins = labelInfo.numPositiveBins + labelInfo.numNegativeBins;
287
288 if (numBins > 0) {
289 // Reset arrays to zero...
290 setArrayToZeros(numElementsPerBin_, numBins);
291
292 // Apply binning method in order to aggregate the gradients and Hessians that belong to the same
293 // bins...
294 typename DenseBinnedScoreVector<IndexVector>::index_binned_iterator binIndexIterator =
295 scoreVector_.indices_binned_begin();
296 auto callback = [=](uint32 binIndex, uint32 labelIndex) {
297 numElementsPerBin_[binIndex] += 1;
298 binIndexIterator[labelIndex] = binIndex;
299 };
300 auto zeroCallback = [=](uint32 labelIndex) {
301 binIndexIterator[labelIndex] = maxBins_;
302 };
303 binningPtr_->createBins(labelInfo, criteria_, numCriteria, callback, zeroCallback);
304
305 // Determine number of non-empty bins...
306 numBins = removeEmptyBins(numElementsPerBin_, binIndices_, numBins);
307 scoreVector_.setNumBins(numBins, false);
308
309 // Aggregate gradients and Hessians...
310 setArrayToZeros(aggregatedGradients_, numBins);
311 setArrayToZeros(aggregatedHessians_, triangularNumber(numBins));
312 aggregateGradientsAndHessians(statisticVector.gradients_cbegin(), statisticVector.hessians_cbegin(),
313 numCriteria, binIndexIterator, binIndices_, aggregatedGradients_,
314 aggregatedHessians_, maxBins_);
315
316 // Copy Hessians to the matrix of coefficients and add regularization weight to its diagonal...
317 copyCoefficients(aggregatedHessians_, this->dsysvTmpArray1_, numBins);
318 addL2RegularizationWeight(this->dsysvTmpArray1_, numBins, numElementsPerBin_,
319 l2RegularizationWeight_);
320
321 // Copy gradients to the vector of ordinates...
322 typename DenseBinnedScoreVector<IndexVector>::score_binned_iterator scoreIterator =
323 scoreVector_.scores_binned_begin();
324 copyOrdinates(aggregatedGradients_, scoreIterator, numBins);
325 addL1RegularizationWeight(scoreIterator, numBins, numElementsPerBin_, l1RegularizationWeight_);
326
327 // Calculate the scores to be predicted for the individual labels by solving a system of linear
328 // equations...
329 lapack_.dsysv(this->dsysvTmpArray1_, this->dsysvTmpArray2_, this->dsysvTmpArray3_, scoreIterator,
330 numBins, this->dsysvLwork_);
331
332 // Calculate the overall quality...
333 float64 quality = calculateOverallQuality(scoreIterator, aggregatedGradients_, aggregatedHessians_,
334 this->dspmvTmpArray_, numBins, blas_);
335
336 // Evaluate regularization term...
337 quality += calculateRegularizationTerm(scoreIterator, numElementsPerBin_, numBins,
338 l1RegularizationWeight_, l2RegularizationWeight_);
339
340 scoreVector_.quality = quality;
341 } else {
342 setArrayToValue(scoreVector_.indices_binned_begin(), numCriteria, maxBins_);
343 scoreVector_.quality = 0;
344 }
345
346 return scoreVector_;
347 }
