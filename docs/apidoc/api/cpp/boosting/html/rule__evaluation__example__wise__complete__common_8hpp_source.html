24 for (uint32 c = 0; c < n; c++) {
25 uint32 offset = c * n;

99 template<typename ScoreIterator, typename GradientIterator, typename HessianIterator>
100 static inline float64 calculateOverallQuality(ScoreIterator scores, GradientIterator gradients,
101 HessianIterator hessians, float64* tmpArray, uint32 numPredictions,
102 const Blas& blas) {
103 blas.dspmv(hessians, scores, tmpArray, numPredictions);
104 return blas.ddot(scores, gradients, numPredictions) + (0.5 * blas.ddot(scores, tmpArray, numPredictions));
105 }

116 template<typename ScoreIterator>
117 static inline float64 calculateRegularizationTerm(ScoreIterator scores, uint32 numPredictions,
118 float64 l1RegularizationWeight, float64 l2RegularizationWeight) {
119 float64 regularizationTerm;
120
121 if (l1RegularizationWeight > 0) {
122 regularizationTerm = l1RegularizationWeight * l1Norm(scores, numPredictions);
123 } else {
124 regularizationTerm = 0;
125 }
126
127 if (l2RegularizationWeight > 0) {
128 regularizationTerm += 0.5 * l2RegularizationWeight * l2NormPow(scores, numPredictions);
129 }
130
131 return regularizationTerm;
132 }

182 const IScoreVector& calculateScores(DenseExampleWiseStatisticVector& statisticVector) override {
183 uint32 numPredictions = scoreVector_.getNumElements();
184
185 // Copy Hessians to the matrix of coefficients and add the L2 regularization weight to its diagonal...
186 copyCoefficients(statisticVector.hessians_cbegin(), this->dsysvTmpArray1_, numPredictions);
187 addL2RegularizationWeight(this->dsysvTmpArray1_, numPredictions, l2RegularizationWeight_);
188
189 // Copy gradients to the vector of ordinates and add the L1 regularization weight...
190 typename DenseScoreVector<IndexVector>::score_iterator scoreIterator = scoreVector_.scores_begin();
191 copyOrdinates(statisticVector.gradients_cbegin(), scoreIterator, numPredictions);
192 addL1RegularizationWeight(scoreIterator, numPredictions, l1RegularizationWeight_);
193
194 // Calculate the scores to be predicted for individual labels by solving a system of linear equations...
195 lapack_.dsysv(this->dsysvTmpArray1_, this->dsysvTmpArray2_, this->dsysvTmpArray3_, scoreIterator,
196 numPredictions, this->dsysvLwork_);
197
198 // Calculate the overall quality...
199 float64 quality = calculateOverallQuality(scoreIterator, statisticVector.gradients_begin(),
200 statisticVector.hessians_begin(), this->dspmvTmpArray_,
201 numPredictions, blas_);
202
203 // Evaluate regularization term...
204 quality += calculateRegularizationTerm(scoreIterator, numPredictions, l1RegularizationWeight_,
205 l2RegularizationWeight_);
206
207 scoreVector_.quality = quality;
