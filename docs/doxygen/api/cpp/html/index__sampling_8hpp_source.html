21static inline void sampleIndicesWithoutReplacementViaTrackingSelection(PartialIndexVector& indexVector,
22 Iterator iterator, uint32 numTotal, RNG& rng) {
23 uint32 numSamples = indexVector.getNumElements();
24 PartialIndexVector::iterator sampleIterator = indexVector.begin();
25 std::unordered_set<uint32> selectedIndices;
26
27 for (uint32 i = 0; i < numSamples; i++) {
28 bool shouldContinue = true;
29 uint32 sampledIndex;
30
31 while (shouldContinue) {
32 uint32 randomIndex = rng.random(0, numTotal);

85template<typename FirstIterator, typename SecondIterator>
86static inline void randomPermutation(FirstIterator firstIterator, SecondIterator secondIterator, uint32 numFirst,
87 uint32 numTotal, uint32 numPermutations, RNG& rng) {
88 for (uint32 i = 0; i < numPermutations; i++) {
89 // Swap elements at index i and at a randomly selected index...
90 uint32 randomIndex = rng.random(i, numTotal);
91 uint32 tmp1 = i < numFirst ? firstIterator[i] : secondIterator[i - numFirst];
92 uint32 tmp2;
93
94 if (randomIndex < numFirst) {
95 tmp2 = firstIterator[randomIndex];
96 firstIterator[randomIndex] = tmp1;
97 } else {
98 tmp2 = secondIterator[randomIndex - numFirst];
99 secondIterator[randomIndex - numFirst] = tmp1;
100 }
101
102 if (i < numFirst) {
103 firstIterator[i] = tmp2;
104 } else {
105 secondIterator[i - numFirst] = tmp2;
106 }
107 }
108}

121static inline void sampleIndicesWithoutReplacementViaRandomPermutation(PartialIndexVector& indexVector,
122 Iterator iterator, uint32 numTotal, RNG& rng) {
123 uint32 numSamples = indexVector.getNumElements();
124 PartialIndexVector::iterator sampleIterator = indexVector.begin();
125 uint32 unusedIndices[numTotal - numSamples];
126
127 for (uint32 i = 0; i < numSamples; i++) {
128 sampleIterator[i] = iterator[i];
129 }
130
131 for (uint32 i = numSamples; i < numTotal; i++) {
132 unusedIndices[i - numSamples] = iterator[i];
133 }
134
135 randomPermutation<PartialIndexVector::iterator, uint32*>(sampleIterator, &unusedIndices[0], numSamples, numTotal,
136 numSamples, rng);
137}

149template<typename Iterator>
150static inline void sampleIndicesWithoutReplacement(PartialIndexVector& indexVector, Iterator iterator, uint32 numTotal,
151 RNG& rng) {
152 float64 ratio = numTotal > 0 ? ((float64) indexVector.getNumElements()) / ((float64) numTotal) : 1;
153
154 // The thresholds for choosing a suitable method are based on empirical experiments
155 if (ratio < 0.06) {
156 // For very small ratios use tracking selection
157 sampleIndicesWithoutReplacementViaTrackingSelection(indexVector, iterator, numTotal, rng);
158 } else if (ratio > 0.5) {
159 // For large ratios use reservoir sampling
160 sampleIndicesWithoutReplacementViaReservoirSampling(indexVector, iterator, numTotal, rng);
161 } else {
162 // Otherwise, use random permutation as the default method
163 sampleIndicesWithoutReplacementViaRandomPermutation(indexVector, iterator, numTotal, rng);
164 }
