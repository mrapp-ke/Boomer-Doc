13 void applyExampleWisePredictionInternally(uint32 statisticIndex, const Prediction& prediction,
14 const LabelMatrix& labelMatrix, StatisticView& statisticView,
15 ScoreMatrix& scoreMatrix, const LossFunction& lossFunction) {
16 // Update the scores that are currently predicted for the example at the given index...
17 scoreMatrix.addToRowFromSubset(statisticIndex, prediction.scores_cbegin(), prediction.scores_cend(),
18 prediction.indices_cbegin(), prediction.indices_cend());
19
20 // Update the gradients and Hessians for the example at the given index...
21 lossFunction.updateExampleWiseStatistics(statisticIndex, labelMatrix, scoreMatrix, statisticView);
22 }

100 void addToMissing(uint32 statisticIndex, float64 weight) override {
101 // Create a vector for storing the totals sums of gradients and Hessians, if necessary...
102 if (totalCoverableSumVector_ == nullptr) {
103 totalCoverableSumVector_ = new StatisticVector(*totalSumVector_);
104 totalSumVector_ = totalCoverableSumVector_;
105 }
106
107 // Subtract the gradients and Hessians of the example at the given index (weighted by the given
108 // weight) from the total sums of gradients and Hessians...
109 totalCoverableSumVector_->add(
110 statistics_.statisticViewPtr_->gradients_row_cbegin(statisticIndex),
111 statistics_.statisticViewPtr_->gradients_row_cend(statisticIndex),
112 statistics_.statisticViewPtr_->hessians_row_cbegin(statisticIndex),
113 statistics_.statisticViewPtr_->hessians_row_cend(statisticIndex), -weight);
114 }

116 void addToSubset(uint32 statisticIndex, float64 weight) override {
117 sumVector_.addToSubset(statistics_.statisticViewPtr_->gradients_row_cbegin(statisticIndex),
118 statistics_.statisticViewPtr_->gradients_row_cend(statisticIndex),
119 statistics_.statisticViewPtr_->hessians_row_cbegin(statisticIndex),
120 statistics_.statisticViewPtr_->hessians_row_cend(statisticIndex),
121 labelIndices_, weight);
122 }
123
124 void resetSubset() override {
125 // Create a vector for storing the accumulated sums of gradients and Hessians, if necessary...
126 if (accumulatedSumVector_ == nullptr) {
127 uint32 numPredictions = labelIndices_.getNumElements();
128 accumulatedSumVector_ = new StatisticVector(numPredictions, true);
129 }
130
131 // Reset the sum of gradients and Hessians to zero and add it to the accumulated sums of
132 // gradients and Hessians...
133 accumulatedSumVector_->add(sumVector_.gradients_cbegin(), sumVector_.gradients_cend(),
134 sumVector_.hessians_cbegin(), sumVector_.hessians_cend());
135 sumVector_.clear();
136 }
137
138 const IScoreVector& calculatePrediction(bool uncovered, bool accumulated) override {
139 StatisticVector& sumsOfStatistics = accumulated ? *accumulatedSumVector_ : sumVector_;
140
141 if (uncovered) {
142 tmpVector_.difference(totalSumVector_->gradients_cbegin(),
143 totalSumVector_->gradients_cend(), totalSumVector_->hessians_cbegin(),
144 totalSumVector_->hessians_cend(), labelIndices_,
145 sumsOfStatistics.gradients_cbegin(),
146 sumsOfStatistics.gradients_cend(), sumsOfStatistics.hessians_cbegin(),
147 sumsOfStatistics.hessians_cend());
148 return ruleEvaluationPtr_->calculatePrediction(tmpVector_);
149 }
150
151 return ruleEvaluationPtr_->calculatePrediction(sumsOfStatistics);
152 }
