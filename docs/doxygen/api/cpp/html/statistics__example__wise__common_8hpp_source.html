13 void applyExampleWisePredictionInternally(uint32 statisticIndex, const Prediction& prediction,
14 const LabelMatrix& labelMatrix, StatisticView& statisticView,
15 ScoreMatrix& scoreMatrix, const LossFunction& lossFunction) {
16 // Update the scores that are currently predicted for the example at the given index...
17 scoreMatrix.addToRowFromSubset(statisticIndex, prediction.scores_cbegin(), prediction.scores_cend(),
18 prediction.indices_cbegin(), prediction.indices_cend());
19
20 // Update the gradients and Hessians for the example at the given index...
21 lossFunction.updateExampleWiseStatistics(statisticIndex, labelMatrix, scoreMatrix, statisticView);
22 }

97 void addToMissing(uint32 statisticIndex, float64 weight) override {
98 // Create a vector for storing the totals sums of gradients and Hessians, if necessary...
99 if (totalCoverableSumVector_ == nullptr) {
100 totalCoverableSumVector_ = new StatisticVector(*totalSumVector_);
101 totalSumVector_ = totalCoverableSumVector_;
102 }
103
104 // Subtract the gradients and Hessians of the example at the given index (weighted by the given
105 // weight) from the total sums of gradients and Hessians...
106 totalCoverableSumVector_->add(
107 statistics_.statisticViewPtr_->gradients_row_cbegin(statisticIndex),
108 statistics_.statisticViewPtr_->gradients_row_cend(statisticIndex),
109 statistics_.statisticViewPtr_->hessians_row_cbegin(statisticIndex),
110 statistics_.statisticViewPtr_->hessians_row_cend(statisticIndex), -weight);
111 }

113 void addToSubset(uint32 statisticIndex, float64 weight) override {
114 sumVector_.addToSubset(statistics_.statisticViewPtr_->gradients_row_cbegin(statisticIndex),
115 statistics_.statisticViewPtr_->gradients_row_cend(statisticIndex),
116 statistics_.statisticViewPtr_->hessians_row_cbegin(statisticIndex),
117 statistics_.statisticViewPtr_->hessians_row_cend(statisticIndex),
118 labelIndices_, weight);
119 }
120
121 void resetSubset() override {
122 // Create a vector for storing the accumulated sums of gradients and Hessians, if necessary...
123 if (accumulatedSumVector_ == nullptr) {
124 uint32 numPredictions = labelIndices_.getNumElements();
125 accumulatedSumVector_ = new StatisticVector(numPredictions, true);
126 }
127
128 // Reset the sum of gradients and Hessians to zero and add it to the accumulated sums of
129 // gradients and Hessians...
130 accumulatedSumVector_->add(sumVector_.gradients_cbegin(), sumVector_.gradients_cend(),
131 sumVector_.hessians_cbegin(), sumVector_.hessians_cend());
132 sumVector_.clear();
133 }
134
135 const IScoreVector& calculatePrediction(bool uncovered, bool accumulated) override {
136 StatisticVector& sumsOfStatistics = accumulated ? *accumulatedSumVector_ : sumVector_;
137
138 if (uncovered) {
139 tmpVector_.difference(totalSumVector_->gradients_cbegin(),
140 totalSumVector_->gradients_cend(), totalSumVector_->hessians_cbegin(),
141 totalSumVector_->hessians_cend(), labelIndices_,
142 sumsOfStatistics.gradients_cbegin(),
143 sumsOfStatistics.gradients_cend(), sumsOfStatistics.hessians_cbegin(),
144 sumsOfStatistics.hessians_cend());
145 return ruleEvaluationPtr_->calculatePrediction(tmpVector_);
146 }
147
148 return ruleEvaluationPtr_->calculatePrediction(sumsOfStatistics);
149 }
