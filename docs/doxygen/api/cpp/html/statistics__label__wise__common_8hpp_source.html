11 template<typename Prediction, typename LabelMatrix, typename StatisticView, typename ScoreMatrix,
12 typename LossFunction>
13 static inline void applyLabelWisePredictionInternally(uint32 statisticIndex, const Prediction& prediction,
14 const LabelMatrix& labelMatrix, StatisticView& statisticView,
15 ScoreMatrix& scoreMatrix, const LossFunction& lossFunction) {
16 // Update the scores that are currently predicted for the example at the given index...
17 scoreMatrix.addToRowFromSubset(statisticIndex, prediction.scores_cbegin(), prediction.scores_cend(),
18 prediction.indices_cbegin(), prediction.indices_cend());
19
20 // Update the gradients and Hessians of the example at the given index...
21 lossFunction.updateLabelWiseStatistics(statisticIndex, labelMatrix, scoreMatrix, prediction.indices_cbegin(),
22 prediction.indices_cend(), statisticView);
23 }

100 void addToMissing(uint32 statisticIndex, float64 weight) override {
101 // Create a vector for storing the totals sums of gradients and Hessians, if necessary...
102 if (totalCoverableSumVector_ == nullptr) {
103 totalCoverableSumVector_ = new StatisticVector(*totalSumVector_);
104 totalSumVector_ = totalCoverableSumVector_;
105 }
106
107 // Subtract the gradients and Hessians of the example at the given index (weighted by the given
108 // weight) from the total sums of gradients and Hessians...
109 totalCoverableSumVector_->add(statistics_.statisticViewPtr_->row_cbegin(statisticIndex),
110 statistics_.statisticViewPtr_->row_cend(statisticIndex), -weight);
111 }

113 void addToSubset(uint32 statisticIndex, float64 weight) override {
114 sumVector_.addToSubset(statistics_.statisticViewPtr_->row_cbegin(statisticIndex),
115 statistics_.statisticViewPtr_->row_cend(statisticIndex), labelIndices_,
116 weight);
117 }
118
119 void resetSubset() override {
120 // Create a vector for storing the accumulated sums of gradients and Hessians, if necessary...
121 if (accumulatedSumVector_ == nullptr) {
122 uint32 numPredictions = labelIndices_.getNumElements();
123 accumulatedSumVector_ = new StatisticVector(numPredictions, true);
124 }
125
126 // Reset the sums of gradients and Hessians to zero and add it to the accumulated sums of
127 // gradients and Hessians...
128 accumulatedSumVector_->add(sumVector_.cbegin(), sumVector_.cend());
129 sumVector_.clear();
130 }

132 const IScoreVector& calculatePrediction(bool uncovered, bool accumulated) override {
133 StatisticVector& sumsOfStatistics = accumulated ? *accumulatedSumVector_ : sumVector_;
134
135 if (uncovered) {
136 tmpVector_.difference(totalSumVector_->cbegin(), totalSumVector_->cend(), labelIndices_,
137 sumsOfStatistics.cbegin(), sumsOfStatistics.cend());
138 return ruleEvaluationPtr_->calculatePrediction(tmpVector_);
139 }
140
141 return ruleEvaluationPtr_->calculatePrediction(sumsOfStatistics);
142 }
