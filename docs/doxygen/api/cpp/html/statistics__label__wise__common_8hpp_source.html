11 template<typename Prediction, typename LabelMatrix, typename StatisticView, typename ScoreMatrix,
12 typename LossFunction>
13 static inline void applyLabelWisePredictionInternally(uint32 statisticIndex, const Prediction& prediction,
14 const LabelMatrix& labelMatrix, StatisticView& statisticView,
15 ScoreMatrix& scoreMatrix, const LossFunction& lossFunction) {
16 // Update the scores that are currently predicted for the example at the given index...
17 scoreMatrix.addToRowFromSubset(statisticIndex, prediction.scores_cbegin(), prediction.scores_cend(),
18 prediction.indices_cbegin(), prediction.indices_cend());
19
20 // Update the gradients and Hessians of the example at the given index...
21 lossFunction.updateLabelWiseStatistics(statisticIndex, labelMatrix, scoreMatrix, prediction.indices_cbegin(),
22 prediction.indices_cend(), statisticView);
23 }

97 void addToMissing(uint32 statisticIndex, float64 weight) override {
98 // Create a vector for storing the totals sums of gradients and Hessians, if necessary...
99 if (totalCoverableSumVector_ == nullptr) {
100 totalCoverableSumVector_ = new StatisticVector(*totalSumVector_);
101 totalSumVector_ = totalCoverableSumVector_;
102 }
103
104 // Subtract the gradients and Hessians of the example at the given index (weighted by the given
105 // weight) from the total sums of gradients and Hessians...
106 totalCoverableSumVector_->add(statistics_.statisticViewPtr_->row_cbegin(statisticIndex),
107 statistics_.statisticViewPtr_->row_cend(statisticIndex), -weight);
108 }

110 void addToSubset(uint32 statisticIndex, float64 weight) override {
111 sumVector_.addToSubset(statistics_.statisticViewPtr_->row_cbegin(statisticIndex),
112 statistics_.statisticViewPtr_->row_cend(statisticIndex), labelIndices_,
113 weight);
114 }
115
116 void resetSubset() override {
117 // Create a vector for storing the accumulated sums of gradients and Hessians, if necessary...
118 if (accumulatedSumVector_ == nullptr) {
119 uint32 numPredictions = labelIndices_.getNumElements();
120 accumulatedSumVector_ = new StatisticVector(numPredictions, true);
121 }
122
123 // Reset the sums of gradients and Hessians to zero and add it to the accumulated sums of
124 // gradients and Hessians...
125 accumulatedSumVector_->add(sumVector_.cbegin(), sumVector_.cend());
126 sumVector_.clear();
127 }

129 const IScoreVector& calculatePrediction(bool uncovered, bool accumulated) override {
130 StatisticVector& sumsOfStatistics = accumulated ? *accumulatedSumVector_ : sumVector_;
131
132 if (uncovered) {
133 tmpVector_.difference(totalSumVector_->cbegin(), totalSumVector_->cend(), labelIndices_,
134 sumsOfStatistics.cbegin(), sumsOfStatistics.cend());
135 return ruleEvaluationPtr_->calculatePrediction(tmpVector_);
136 }
137
138 return ruleEvaluationPtr_->calculatePrediction(sumsOfStatistics);
139 }
